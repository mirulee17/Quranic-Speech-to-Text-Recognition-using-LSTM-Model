# Constants

MAX_LEN = 50

@app.route('/')
def home():
    return '✅ Flask Quranic Verse Classifier is Running', 200

def predict_verse(audio_np):
    try:
        y_trimmed, _ = librosa.effects.trim(audio_np, top_db=25)
        if len(y_trimmed) == 0:
            return None, 0.0

        mfcc = librosa.feature.mfcc(y=y_trimmed, sr=RATE, n_mfcc=N_MFCC).T
        if mfcc.shape[0] > MAX_LEN:
            mfcc = mfcc[:MAX_LEN]
        else:
            mfcc = np.pad(mfcc, ((0, MAX_LEN - mfcc.shape[0]), (0, 0)))

        mfcc = (mfcc - np.mean(mfcc)) / (np.std(mfcc) + 1e-10)
        mfcc = np.expand_dims(mfcc, axis=0)

        prediction = model.predict(mfcc, verbose=0)
        predicted_index = int(np.argmax(prediction)) + 1
        confidence = float(np.max(prediction))
        return predicted_index, confidence
    except Exception as e:
        logging.error(f"❌ Prediction Error: {str(e)}")
        return None, 0.0

@app.route('/predict', methods=['POST'])
def predict():
    try:
        audio_base64 = request.form.get("audio_base64", "")
        surah = request.form.get("surah", "al-ikhlas").strip().lower()

        if not audio_base64:
            logging.warning("⚠️ Missing audio_base64 field")
            return jsonify({"error": "Missing 'audio_base64' field"}), 400

        try:
            audio_bytes = base64.b64decode(audio_base64)
            audio_np = np.frombuffer(audio_bytes, dtype=np.int16).astype(np.float32) / 32768.0
        except Exception as e:
            logging.warning("⚠️ Base64 decode failed")
            return jsonify({"error": f"Base64 decoding error: {str(e)}"}), 400

        verse_index, confidence = predict_verse(audio_np)

        if verse_index is None:
            return jsonify({"message": "Silence or unreadable audio"}), 200

        if confidence > 0.4:
            verse_text = verse_texts_by_surah.get(surah, {}).get(verse_index, "Unknown verse")
            logging.info(f"✅ Predicted: Verse {verse_index}, Confidence {confidence:.2f}")
            return jsonify({
                "verse_index": verse_index,
                "confidence": round(confidence, 4),
                "verse_text": verse_text
            })
        else:
            logging.info(f"⚠️ Low Confidence: {confidence:.2f}")
            return jsonify({
                "message": "Low confidence",
                "confidence": round(confidence, 4)
            }), 200

    except Exception as e:
        logging.exception("❌ Backend Exception:")
        return jsonify({"error": "Internal server error", "details": str(e)}), 500

if __name__ == '__main__':
